#Definindo as colunas que irei analisar.
colunas = ['Attrition_Flag', 'Gender', 'Education_Level', 'Marital_Status', 'Income_Category', 'Card_Category']

#Criando a funo
def pltCountplot(colunas):
    fig, axis = plt.subplots(len(colunas) // 3,3, figsize=(20,12))  

    index = 0
    for i in range(len(colunas) // 3):
        for j in range(3):
            
            ax = sns.countplot(colunas[index], data=df, ax=axis[i][j]) #Sempre que houver um incremento a coluna ser
            #alterada
            if colunas[index] in ['Education_Level', 'Income_Category']: #Como essas colunas tem mais dados
                for item in ax.get_xticklabels(): #sempre que elas aparecerem os seus valores em x sero rotacionados
                    item.set_rotation(15) #em 15
                
            for p in ax.patches:
                height = p.get_height()
                ax.text(p.get_x()+p.get_width()/2.,
                        height + 3,
                        '{:1.2f}%'.format(height/len(df)*100),
                        ha="center") 
            index += 1

    dv = results.std()
    print('Acurcia mdia: {:.2f}%'.format(mean*100))
